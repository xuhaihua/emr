verticalBottomInterval

verticalBottomInterval属性用于设置子节点垂直对齐时：结束节点底部是否带有对齐间隔

  • verticalBottomInterval = true

<>
    <!--
        创建一个平面容器，设置contentVertical属性为bottom控制其中的子元素靠底部对齐
        设置verticalInterval属性为20控制各节点之间的间隔为20mm
        设置verticalBottomInterval属性为true使结束节点底部存在对齐间隔
     -->
    <>

        <!--子节点-->
        <></PanelLayer>
        <></PanelLayer>
        <></PanelLayer>
    </PanelRoot>
</Root>

  • verticalBottomInterval = false

<>
    <!--
        创建一个平面容器，设置contentVertical属性为bottom控制其中的子元素靠底部对齐
        设置verticalInterval属性为20控制各节点之间的间隔为20mm
        设置verticalBottomInterval属性为true取消开始节点正后侧的对齐间隔
    -->
    <>

        <!--子节点-->
        <></PanelLayer>
        <></PanelLayer>
        <></PanelLayer>
    </PanelRoot>
</Root>
