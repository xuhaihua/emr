forwardBackInterval

forwardBackInterval属性用于设置子节点Z轴方向对齐时：开始节点正后侧是否带有对齐间隔

  • forwardBackInterval = true

<>
    <!--
        创建一个空间节点，设置contentForward属性为back控制其中的子元素靠Z轴正后方对齐
        设置forwardInterval属性为20控制各节点之间的间隔为20mm
        设置forwardBackInterval属性为true使开始节点正后侧存在对齐间隔
     -->
    <>

        <!--子节点-->
        <></SpaceNode>
        <></SpaceNode>
        <></SpaceNode>
    </SpaceNode>
</Root>

  • forwardBackInterval = false

<>
    <!--
        创建一个空间节点，设置contentForward属性为back控制其中的子元素靠Z轴正后方对齐
        设置forwardInterval属性为20控制各节点之间的间隔为20mm
        设置forwardBackInterval属性为false取消开始节点正后侧的对齐间隔
    -->
    <>

        <!--子节点-->
        <></SpaceNode>
        <></SpaceNode>
        <></SpaceNode>
    </SpaceNode>
</Root>
